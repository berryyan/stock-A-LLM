# React Project Initialization Commands
# Copy and paste these commands one by one in Anaconda Prompt

# 1. Create Vite project (when prompted, choose: y to overwrite)
npm create vite@latest . -- --template react-ts

# 2. Install base dependencies
npm install

# 3. Install Markdown and syntax highlighting
npm install react-markdown remark-gfm rehype-katex remark-math katex
npm install react-syntax-highlighter @types/react-syntax-highlighter

# 4. Install routing and utilities
npm install axios react-router-dom clsx

# 5. Install Tailwind CSS and dev tools
npm install -D tailwindcss postcss autoprefixer @tailwindcss/typography
npm install -D prettier eslint-config-prettier

# 6. Initialize Tailwind CSS
npx tailwindcss init -p

# 7. Test the setup
npm run dev
